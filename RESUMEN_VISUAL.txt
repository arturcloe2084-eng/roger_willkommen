EXPORTAR:
  • Como JSON completo      → descarga diccionario-completo.json
  • Solo personalizadas      → descarga diccionario-personalizado.json
  • Como CSV                 → descarga diccionario.csv

IMPORTAR:
  • CSV                      → Directo en herramienta web
  • JSON                     → Via DictionaryHelper
  • Palabras de inicio       → DictionaryHelper.loadStarterWords()

SINCRONIZAR:
  • Juego ←→ localStorage   → Automático, sin acción
  • Backup                  → Exporta regularmente

LIMPIAR:
  • Eliminar custom         → localStorage.removeItem('customDictionaryWords')
  • Eliminar todo           → localStorage.clear()

import { DictionaryManager } from '../../services/DictionaryManager.js';

// En la escena
async create() {
  const dict = await DictionaryManager.getInstance();
  
  // Obtener todas las palabras
  const all = dict.getAll();
  
  // Obtener por categoría
  const saludos = dict.getByCategory('saludo');
  
  // Buscar
  const results = dict.search('Hallo');
  
  // Agregar (en juego también)
  dict.addWord('Danke', 'Gracias', 'cortesía', 'Danke schön!');
}
